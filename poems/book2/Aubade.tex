\settowidth{\versewidth}{Heart pokes heart. Love never ends,}
\begin{verse}[\versewidth]
    Do I love her? That depends\\*
    \vin On when \& wherefore thighs were parted.\\
    Heart pokes heart. Love never ends,\\*
    \vin Unless it never started.
\end{verse}

\blfootnote{Having heard the woman speaker of Dowland's famous parting-at-dawn poem -- the one beginning, `Stay, O sweet, and do not rise' -- say her part, the man confers with his soul.}
